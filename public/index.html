  <script>

    const LANGUAGES = [{ code: "en", label: "🇬🇧 English" }, { code: "ru", label: "🇷🇺 Русский" }, { code: "uk", label: "🇺🇦 Українська" }];

    const TONE_EN = ["Formal", "Neutral", "Casual", "Friendly"];

    const LEVEL_EN = ["Simple", "Standard", "Advanced"];


    const UI = {

      en: { title: "Proofreader", subtitle: "Paste your text and get it polished instantly", inputLabel: "Your text", outputLabel: "Proofread result", inputPlaceholder: "Type or paste your text here...", outputPlaceholder: "Result will appear here...", proofread: "Proofread", tone: "Tone", level: "Language Level", loading: "Proofreading...", language: "Language", settings: "Settings", tones: ["Formal", "Neutral", "Casual", "Friendly"], levels: ["Simple", "Standard", "Advanced"] },

      ru: { title: "Пруфридер", subtitle: "Вставьте текст и получите исправленную версию мгновенно", inputLabel: "Ваш текст", outputLabel: "Результат", inputPlaceholder: "Введите или вставьте текст сюда...", outputPlaceholder: "Результат появится здесь...", proofread: "Проверить", tone: "Тон", level: "Уровень языка", loading: "Проверяем...", language: "Язык", settings: "Настройки", tones: ["Официальный", "Нейтральный", "Непринуждённый", "Дружелюбный"], levels: ["Простой", "Стандартный", "Продвинутый"] },

      uk: { title: "Пруфрідер", subtitle: "Вставте текст і отримайте виправлену версію миттєво", inputLabel: "Ваш текст", outputLabel: "Результат", inputPlaceholder: "Введіть або вставте текст сюди...", outputPlaceholder: "Результат з'явиться тут...", proofread: "Перевірити", tone: "Тон", level: "Рівень мови", loading: "Перевіряємо...", language: "Мова", settings: "Налаштування", tones: ["Офіційний", "Нейтральний", "Невимушений", "Дружній"], levels: ["Простий", "Стандартний", "Просунутий"] }

    };


    const SYSTEM_PROMPT = {

      en: (tone, level) => `You are a professional proofreader. Proofread and correct the user's text. Tone: ${tone}. Language level: ${level}. Return ONLY the corrected text in English, no explanations or comments.`,

      ru: (tone, level) => `Ты профессиональный корректор. Проверь и исправь текст пользователя, верни исправленную версию на русском языке. Тон: ${tone}. Уровень языка: ${level}. Верни ТОЛЬКО исправленный текст на русском языке, без объяснений.`,

      uk: (tone, level) => `Ти професійний коректор. Перевір та виправ текст користувача, поверни виправлену версію українською мовою. Тон: ${tone}. Рівень мови: ${level}. Поверни ЛИШЕ виправлений текст українською мовою, без пояснень.`,

    };


    let state = {

      lang: localStorage.getItem("pr_lang") || "en",

      toneIdx: parseInt(localStorage.getItem("pr_tone")) || 1,

      levelIdx: parseInt(localStorage.getItem("pr_level")) || 1,

      input: localStorage.getItem("pr_input") || "",

      output: localStorage.getItem("pr_output") || "",

      loading: false

    };


    const els = {

      input: document.getElementById('input-text'), output: document.getElementById('output-text'), btn: document.getElementById('proofread-btn'),

      langGroup: document.getElementById('lang-group'), levelGroup: document.getElementById('level-group'), toneGroup: document.getElementById('tone-group'),

      copyInput: document.getElementById('copy-input'), copyOutput: document.getElementById('copy-output')

    };


    function renderText() {

      const t = UI[state.lang];

      document.getElementById('t-title').innerText = t.title; document.getElementById('t-subtitle').innerText = t.subtitle;

      document.getElementById('t-settings').innerText = t.settings; document.getElementById('t-lang-label').innerText = t.language;

      document.getElementById('t-level-label').innerText = t.level; document.getElementById('t-tone-label').innerText = t.tone;

      document.getElementById('t-input-label').innerText = t.inputLabel; document.getElementById('t-output-label').innerText = t.outputLabel;

      els.input.placeholder = t.inputPlaceholder; els.btn.innerText = state.loading ? t.loading : t.proofread;

      

      if (!state.output) { els.output.innerText = state.loading ? t.loading : t.outputPlaceholder; els.output.classList.remove('has-content'); } 

      else { els.output.innerText = state.output; els.output.classList.add('has-content'); }

    }


    function renderChips() {

      const t = UI[state.lang];

      els.langGroup.innerHTML = LANGUAGES.map(l => `<button class="chip ${state.lang === l.code ? 'active' : ''}" data-type="lang" data-val="${l.code}">${l.label}</button>`).join('');

      els.levelGroup.innerHTML = t.levels.map((lvl, i) => `<button class="chip ${state.levelIdx === i ? 'active' : ''}" data-type="level" data-val="${i}">${lvl}</button>`).join('');

      els.toneGroup.innerHTML = t.tones.map((tn, i) => `<button class="chip ${state.toneIdx === i ? 'active' : ''}" data-type="tone" data-val="${i}">${tn}</button>`).join('');

    }


    function updateState(key, val) { state[key] = val; localStorage.setItem(`pr_${key}`, val); updateUI(); }

    function updateUI() { renderText(); renderChips(); els.input.value = state.input; autoResize(els.input); els.btn.disabled = state.loading || !state.input.trim(); els.copyOutput.disabled = !state.output; }

    function autoResize(el) { el.style.height = 'auto'; el.style.height = Math.max(el.scrollHeight, 90) + 'px'; }


    // --- ВОТ ГЛАВНОЕ ИЗМЕНЕНИЕ: ОТПРАВЛЯЕМ ЗАПРОС НА VERCEL API ---

    async function handleProofread() {

      if (!state.input.trim() || state.loading) return;

      

      updateState('loading', true);

      updateState('output', '');

      

      const promptText = `${SYSTEM_PROMPT[state.lang](TONE_EN[state.toneIdx], LEVEL_EN[state.levelIdx])}\n\nOriginal text:\n${state.input}`;


      try {

        const res = await fetch('/api/proofread', {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify({ prompt: promptText })

        });

        

        const data = await res.json();

        if (!res.ok) throw new Error(data.error || 'Server error');

        

        updateState('output', data.result);

      } catch (err) {

        updateState('output', "Error: " + err.message);

      } finally {

        updateState('loading', false);

      }

    }


    els.input.addEventListener('input', (e) => { updateState('input', e.target.value); });

    els.input.addEventListener('keydown', (e) => { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); handleProofread(); } });

    els.btn.addEventListener('click', handleProofread);

    document.addEventListener('click', (e) => {

      if (e.target.classList.contains('chip')) {

        const type = e.target.getAttribute('data-type'); const val = e.target.getAttribute('data-val');

        if (type === 'lang') updateState('lang', val); if (type === 'level') updateState('levelIdx', parseInt(val)); if (type === 'tone') updateState('toneIdx', parseInt(val));

      }

    });


    function copyText(text, btnId, svgId) {

      if (!text) return; navigator.clipboard.writeText(text);

      const btn = document.getElementById(btnId); const svg = document.getElementById(svgId);

      btn.classList.add('copied'); svg.innerHTML = `<polyline points="2,8 6,12 14,4" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" fill="none" />`;

      setTimeout(() => { btn.classList.remove('copied'); svg.innerHTML = `<rect x="5" y="1" width="9" height="11" rx="1.5" /><path d="M2 5h0a1.5 1.5 0 0 0-1.5 1.5v8A1.5 1.5 0 0 0 2 16h6a1.5 1.5 0 0 0 1.5-1.5v0" />`; }, 1500);

    }


    els.copyInput.addEventListener('click', () => copyText(state.input, 'copy-input', 'svg-input'));

    els.copyOutput.addEventListener('click', () => copyText(state.output, 'copy-output', 'svg-output'));


    updateUI();

  </script>
